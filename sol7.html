Using List Comprehension:

python
Copy code
def intersection(list1, list2):
    return [value for value in list1 if value in list2]

# Example usage:
list1 = [1, 2, 3, 4, 5]
list2 = [4, 5, 6, 7, 8]
print("Intersection of lists:", intersection(list1, list2))

def intersection(list1, list2):
    set1 = set(list1)
    set2 = set(list2)
    return list(set1.intersection(set2))

# Example usage:
list1 = [1, 2, 3, 4, 5]
list2 = [4, 5, 6, 7, 8]
print("Intersection of lists:", intersection(list1, list2))
