def rod_cutting(length, prices):
            n = len(prices)
            # Create a table to store maximum revenue for rod lengths from 0 to n
            revenue = [0] * (length + 1)
        
            # Iterate over each rod length from 1 to length
            for i in range(1, length + 1):
                # Initialize max revenue for the current length as negative infinity
                max_revenue = float('-inf')
                # Iterate over each possible cut length from 1 to min(i, n)
                for j in range(1, min(i + 1, n)):
                    # Calculate revenue for the current cut length and update max revenue
                    max_revenue = max(max_revenue, prices[j] + revenue[i - j])
                # Store the maximum revenue for the current rod length
                revenue[i] = max_revenue
        
            # Return the maximum revenue obtainable for the given rod length
            return revenue[length]
        
        # Example usage:
        length = 8
        prices = [0, 1, 5, 8, 9, 10, 17, 17, 20]  # Price table for rod lengths from 1 to 8
        max_revenue = rod_cutting(length, prices)
        print("Maximum revenue obtainable by cutting the rod:", max_revenue)
